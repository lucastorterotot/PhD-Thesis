\frametitle{Particles isolation}
\begin{block}{}
\begin{equation*}
I_\text{rel}^{(i)}
=
\left.
\frac{1}{p_T^{(i)}}
\left[
\sum_{\hadron^\pm,\text{PV}} p_T^{\hadron^\pm}
+
\max\left(
0
,
\sum_{\hadron^0}p_{\hadron^0}
+
\sum_{\gamma}p_{\gamma}
- \Delta\beta
\sum_{\hadron^\pm,\text{PU}} p_T^{\hadron^\pm}
\right)
\right]
\right|_{\Delta R < R_i}
\end{equation*}
\end{block}
\begin{center}
$I_\text{rel}<\num{0.3}$,
\qquad
$\Delta R = \sqrt{\Delta\phi^2+\Delta\eta^2}$
\end{center}
