\im \bar{\psi} \slashed{D} \psi

\begin{align}

\mend[,]\\
\mathcal{F}_{\mu\nu}\mathcal{F}^{\mu\nu} &= \bm{G}_{\mu\nu} \cdot \bm{G}^{\mu\nu} + \bm{W}_{\mu\nu} \cdot \bm{W}^{\mu\nu} + F^{(B)}_{\mu\nu} F^{(B)\mu\nu}
\mend[,]\\
V(\phi) &= \mu^2 \phi^\dagger\phi + \lambda (\phi^\dagger\phi)^2
\mend
\end{align}
